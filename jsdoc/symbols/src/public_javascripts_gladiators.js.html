  4 /**
  5  * Event when tile is clicked
  6  * @param {Object} oSource - source element of click
  7  */
  8 function onClickTile(oSource) {
  9     let x = oSource.getAttribute("x"),
 10         y = oSource.getAttribute("y");
 11     if (oCurrGladiator.source === "shop") {
 12         //buy
 13         sendBuyRequest(parseInt(oCurrGladiator.shopIndex)+1, parseInt(x), parseInt(y), function(oEvent) {
 14             updateGame();
 15             if (JSON.stringify(oEvent.player) === JSON.stringify(oController.playerOne)) {
 16                 animateValue("idPlayer1Credits", oEvent.player.credits, 1000);
 17             } else {
 18                 animateValue("idPlayer2Credits", oEvent.player.credits, 1000);
 19             }
 20         });
 21         resetCurrGladiator();
 22     } else if(oCurrGladiator.source === "board") {
 23         //move
 24         let oGladiator = oCurrGladiator.gladiatorDiv.data("gladiator");
 25         sendMoveRequest(oGladiator.position.x, oGladiator.position.y, parseInt(x), parseInt(y), function(oEvent) {
 26             if (oEvent.eventType === "Moved") {
 27                 // moved animation
 28                 oCurrGladiator.gladiatorDiv.data("gladiator", oEvent.gladiator)
 29                 animateGladiatorMove(oCurrGladiator.gladiatorDiv.parent(), $("#idTileX"+x+"Y"+y), 1500);
 30             } else if (oEvent.eventType === "Mined") {
 31                 // mined animation
 32                 updateBoard();
 33             } else if (oEvent.eventType === "BaseAttacked" || oEvent.eventType === "Won") {
 34                 if (JSON.stringify(oEvent.currentPlayer) === JSON.stringify(oController.playerTwo)) {
 35                     animateValue("idPlayer1Health", oController.playerOne.health, 1000);
 36                 } else {
 37                     animateValue("idPlayer2Health", oController.playerTwo.health, 1000);
 38                 }
 39             }
 40             resetCurrGladiator();
 41         });
 42     } else {
 43         resetCurrGladiator();
 44     }
 45 }

 47 /**
 48  * Event when gladiator is clicked
 49  * @param {Object} e - click event
 50  */
 51 function onClickGladiator(e) {
 52     e.stopPropagation();
 53     let oClickedGladiator = $(e.target).data("gladiator");
 54     if(oCurrGladiator.source === "board") {
 55         //attack
 56         let oSrcGladiator = oCurrGladiator.gladiatorDiv.data("gladiator");
 57         sendMoveRequest(oSrcGladiator.position.x, oSrcGladiator.position.y, oClickedGladiator.position.x, oClickedGladiator.position.y, function(oEvent) {
 58             let oParent = $(e.target).parent(),
 59                 bFound = false;
 60             animateGladiatorAttack(oParent);
 61             // get attacked gladiator FIXME: give back in event from backend
 62             oController.playerOne.gladiators.concat(oController.playerTwo.gladiators).forEach(function(gladiator) {
 63                 if (JSON.stringify(gladiator.position) == JSON.stringify(oClickedGladiator.position)) {
 64                     $("#idTileX"+gladiator.position.x+"Y"+gladiator.position.y).find(".gladiator").data("gladiator", gladiator);
 65                     updateHealthBar(oParent);
 66                     bFound = true;
 67                 }
 68             });
 69             //if (oEvent.killed) { FIXME: oEvent.killed not working yet              
 70             if (!bFound) {
 71                 updateHealthBar(oParent, true);
 72                 setTimeout(function() {$(e.target).parent().remove()}, 1000); 
 73             }
 74             resetCurrGladiator();
 75         });
 76     } else {
 77         if (oClickedGladiator.moved === false && 
 78             ((JSON.stringify(oController.currentPlayer) === JSON.stringify(oController.playerOne) && $(e.target).hasClass("glad-player1"))
 79             || (JSON.stringify(oController.currentPlayer) === JSON.stringify(oController.playerTwo) && $(e.target).hasClass("glad-player2")))) {
 80 
 81             toggleActiveClass(e.target);
 82             sendRequest("POST","/gladiators/api/gladiatorSelect", {"x": oClickedGladiator.position.x, "y": oClickedGladiator.position.y}, function(oResult) {
 83                 highlightMoveTiles(oResult.tilesMove);
 84                 highlightAttackTiles(oResult.tilesAttack);
 85             });
 86             oCurrGladiator = { 
 87                 gladiatorDiv: $(e.target),
 88                 source: "board"
 89             }
 90         } else {
 91             animateGladiatorShake($(e.target));
 92         }
 93     }
 94 }

261 /**

265 * @param {int} iNewX - x-value of new position
266 * @param {int} iNewY - y-value of new position

269 function sendMoveRequest(iX, iY, iNewX, iNewY, fnSuccess) {

279 /**
280  * Sends a request to the server
281  * @param {string} sMethod - The method of the HTTP request
282  * @param {string} sPath - The request URL
283  * @param {object} oPayload - The request body dada
284  * @param {function} fnSuccess - Callback function for success
285  */
286 function sendRequest(sMethod, sPath, oPayload, fnSuccess) {
287     $.ajax({
288         method: sMethod,
289         url: sPath,
290         data: JSON.stringify(oPayload),
291         dataType: "json",
292         contentType: "application/json",
293         success: function (oResult) {
294             oController = oResult[0];
295             if (fnSuccess) {
296                 fnSuccess(oResult[1]);
297             }
298         }.bind(this),
299         error: function(oResponse) {
300             resetCurrGladiator();
301             Msg.error(oResponse.responseJSON.message, 2000);
302         }
303     });
304 }

334 /**
335  * Updates the healthbar of a gladiator
336  * @param {Object} oGladiatorContainerDiv - div that contains gladiator and healthbar

339 function updateHealthBar(oGladiatorContainerDiv, bKilled) {
340     if (!bKilled) {
341         let oGladiator = oGladiatorContainerDiv.find(".gladiator").data("gladiator"),
342             iPercentage = oGladiator.healthPoints / oGladiator.initialHealthPoints * 100;
343         oGladiatorContainerDiv.find(".healthbar-inside").css("width", iPercentage+"%");
344         oGladiatorContainerDiv.find(".healthbar-inside").css("background", perc2color(iPercentage));
345     } else {
346         oGladiatorContainerDiv.find(".healthbar-inside").css("width", 0);
347         oGladiatorContainerDiv.find(".healthbar-inside").css("background", perc2color(0));
348     }

351 /**
352  * Returns a color for a percentage between green and red
353  * https://gist.github.com/mlocati/7210513
354  * @param {int} iPercentage - percentage
355  */
356 function perc2color(iPercentage) {
357 	let r, g, b = 0;
358 	if(iPercentage < 50) {
359 		r = 255;
360 		g = Math.round(5.1 * iPercentage);
361 	}
362 	else {
363 		g = 255;
364 		r = Math.round(510 - 5.10 * iPercentage);
365 	}
366 	let h = r * 0x10000 + g * 0x100 + b * 0x1;
367 	return '#' + ('000000' + h.toString(16)).slice(-6);
368 }

385  * @param {String} sElementClass - class of elements

388 function removePrefixClass(sElementClass, sRemoveClass) {
389     $("."+sElementClass).each(function() {

396 /**
397  * Animates the change of an integer value
398  * @param {String} sId - id of element
399  * @param {Integer} iNewValue - new value
400  * @param {Integer} iDuration - duration time of animation in ms
401  */
402 function animateValue(sId, iNewValue, iDuration) {
403     let iStart = parseInt(document.getElementById(sId).innerHTML);
404     if (iStart === iNewValue) return;
405     let range = iNewValue - iStart;
406     let current = iStart;
407     let increment = iNewValue > iStart? 1 : -1;
408     let stepTime = Math.abs(Math.floor(iDuration / range));
409     let obj = document.getElementById(sId);
410     let timer = setInterval(function() {
411         current += increment;
412         obj.innerHTML = current;
413         if (current == iNewValue) {
414             clearInterval(timer);
415         }
416     }, stepTime);
417 }

419 /**
420  * Animates the movement of an element
421  * @param {String} oMoveElement - Element to be moved
422  * @param {Integer} oNewParent - new value
423  * @param {Integer} iDuration - duration time of animation in ms
424  */
425 function animateGladiatorMove(oMoveElement, oNewParent, iDuration) {
426     let newEle = oMoveElement.clone(true).appendTo(oNewParent),
427         newPos = newEle.position();
428     newEle.hide();
429     oMoveElement.find(".healthbar").hide();
430     oMoveElement.css('position', 'absolute').animate(newPos, iDuration, function() {
431         newEle.show();
432         oMoveElement.remove();
433     });
434     return newEle;
435 };
