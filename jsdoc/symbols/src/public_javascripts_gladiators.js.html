144 /**
145  * Sends a request to the server
146  * @param {string} sPath - The request URL
147  * @param {function} fnSuccess - Callback function for success
148  */
149 function sendRequest(sPath, fnSuccess) {
150     $.ajax({
151         method: "GET",
152         url: sPath,
153         dataType: "json",
154         success: function (oResult) {
155             //save Json
156             oController = oResult;
157             if (fnSuccess) {
158                 fnSuccess(oResult);
159             }
160             //FIXME: not always call this, put this into fnSuccess callback
161             updateGladiators(); 
162             updateShop();
163         }.bind(this),
164         error: function(oResult) {
165             console.log(oResult);
166         }
167     });
168 }

170 /**
171  * Sets the active class for an element
172  * Removes active class from the rest
173  * @param {Object} oElement - the new active element
174  */
175 function toggleActiveClass(oElement) {
176     $(".overlay").removeClass("active");
177     $(".shop-item").removeClass("active");
178     if (oElement) {
179         $(oElement).addClass("active");
180     }
181 }

183 /**
184  * Removes a class from elements
185  * @param {String} sElementClass - class of elements
186  * @param {String} sRemoveClass - to be removed class-prefix
187  */
188 function removeClass(sElementClass, sRemoveClass) {
189     $("."+sElementClass).removeClass (function (index, className) {
190         return (className.match ("/(^|\s)"+sRemoveClass"-\S+/g") || []).join(' ');
191     });
192 }
